import pandas as pd
import numpy as np
import plotly.graph_objects as go
import plotly.io as pio

df = pd.read_csv("data/df_enriched.csv")

target_var = "life_expectancy"

predictors = [
    "health_expenditure_usd",
    "physicians_per_1000",
    "nurses_per_1000",
    "smoking_prevalence",
    "diabetes_prevalence",
    "pollution_mortality_rate",
]

# Use log-scale on x for these predictors (you can add more if you want)
log_x_predictors = {"health_expenditure_usd"}

fig = go.Figure()
buttons = []

for i, pred in enumerate(predictors):
    sub = df[[pred, target_var]].dropna()
    x = sub[pred]
    y = sub[target_var]

    # Scatter points (one trace per predictor)
    fig.add_trace(
        go.Scatter(
            x=x,
            y=y,
            mode="markers",
            name=f"{pred}",
            marker=dict(size=7, opacity=0.7),
            visible=(i == 0),
        )
    )

    # Simple linear fit line
    m, b = np.polyfit(x, y, 1)
    x_line = np.linspace(x.min(), x.max(), 100)
    y_line = m * x_line + b

    fig.add_trace(
        go.Scatter(
            x=x_line,
            y=y_line,
            mode="lines",
            name=f"{pred} trend",
            line=dict(dash="dash"),
            visible=(i == 0),
        )
    )

    # Visibility mask (2 traces per predictor)
    vis = [False] * (2 * len(predictors))
    vis[2 * i] = True       # points
    vis[2 * i + 1] = True   # trend line

    pretty_name = pred.replace("_", " ").title()
    xaxis_type = "log" if pred in log_x_predictors else "linear"

    buttons.append(
        dict(
            label=pretty_name,
            method="update",
            args=[
                {"visible": vis},
                {
                    "title": f"{pretty_name} vs Life Expectancy",
                    "xaxis": {
                        "title": pretty_name,
                        "type": xaxis_type,
                        "showgrid": True,
                        "zeroline": False,
                    },
                    "yaxis": {
                        "title": "Life Expectancy (years)",
                        "showgrid": True,
                        "zeroline": False,
                    },
                },
            ],
        )
    )

# Initial axis type based on first predictor
init_xaxis_type = "log" if predictors[0] in log_x_predictors else "linear"
init_pretty = predictors[0].replace("_", " ").title()

fig.update_layout(
    title=f"{init_pretty} vs Life Expectancy",
    xaxis=dict(
        title=init_pretty,
        type=init_xaxis_type,
        showgrid=True,
        zeroline=False,
    ),
    yaxis=dict(
        title="Life Expectancy (years)",
        showgrid=True,
        zeroline=False,
    ),
    updatemenus=[
        dict(
            type="dropdown",
            x=0.5,
            xanchor="center",
            y=1.15,
            yanchor="top",
            buttons=buttons,
            showactive=True,
        )
    ],
    margin=dict(l=70, r=20, t=80, b=70),
)

pio.renderers.default = "iframe"
